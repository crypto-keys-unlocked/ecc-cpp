235 BigInt randNum;
236 size_t numBytes = (numBits + 7) / 8;

238 #if defined(_WIN32) || defined(_WIN64)
239 // std:: cout <<" Entering here \n";
240 HCRYPTPROV hProvider = 0;
241 BYTE* buffer = new BYTE[numBytes];
242 if (CryptAcquireContext(&hProvider, NULL, NULL, PROV_RSA_FULL, CRYPT_VERIFYCONTEXT)) {
243 CryptGenRandom(hProvider, numBytes, buffer);
244 mpz_import(randNum.value, numBytes, 1, sizeof(buffer[0]), 0, 0, buffer);
245 CryptReleaseContext(hProvider, 0);
246 }
247 delete[] buffer;
248
249 #elif defined(__linux__) || defined(__unix__) || defined(__APPLE__)
250 int fd = open("/dev/urandom", O_RDONLY);
251 if (fd != -1) {
252 char* buffer = new char[numBytes];
253 read(fd, buffer, numBytes);
254 mpz_import(randNum.value, numBytes, 1, sizeof(buffer[0]), 0, 0, buffer);
255 close(fd);
256 delete[] buffer;
257 }

261 gmp_randinit_default(state);
262 mpz_urandomb(randNum.value, state, numBits);

279std::string bigIntToString(const BigInt& bigint) {
280 std::string result;
281 BigInt current = bigint;
282 BigInt zero(static_cast<unsigned long int>(0));
283 const BigInt byteSize(static_cast<unsigned long int>(256));
284
285 while (current > zero) {
286 try {
287 unsigned long remainderValue = (current % byteSize).toULongSafe();
288 unsigned char byte = static_cast<unsigned char>(remainderValue);
289 result.insert(0, 1, byte);
290 current /= byteSize;
291 } catch (const std::overflow_error& e) {
292 break;
293 }
294 }
295
296 return result;
